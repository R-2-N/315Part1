<statement_list> ::= <statement> ";" <statement_list> | <statement> ";"
<statement> ::= <void_statement> | <value_statement> | <comment>
<comment> ::= "//" <text>

<void_statement> ::= <conditional_statement> | <loop> | <function_def>
<value_statement> ::= <function_call> | <expression> | <integer> | <float> | <string> | <boolean> | <not> <value_statement>

<conditional_statement> ::= <if> <value_statement> <do> "{" <statement_list> "}" | <if> <conditional_expression> <do> "{" <statement_list> "}" <else> <do> "{" <statement_list> "}"

<loop> ::= <while_loop> | <for_loop>
<while_loop> ::= <while> <value_statement> <do> "{" <statement_list> "}" 
<for_loop> ::= <for> <assignment_expression> ";" <value_statement> ";" <arithmetic_expression> "do {" <statement_list> "}" 

<function_def> ::= <def> <variable> "(" <variable_list> ")" <do> "{" <statement_list> <return> "(" <value_statement> ")" "}"
<function_call> ::= <call> <string> "(" <variable_list> ")" | <call> <function_primitive>
<variable_list> ::= <variable> "," <variable_list> | <variable>
<function_primitive> ::= "getHeading()" | "getAltitude()" | "getTime()" | "ascend()" | "descend()" | "verticalStop()" | "moveForward()" | "moveBackward()" | "horizontalStop()" | "turnLeft1Deg()" | "turnRight1Deg()" | "enableSpray()" | "disableSpray()"

<expression> ::= <conditional_expresion> | <assignment_expression> | <arithmetic_expression>

<conditional_expression> ::= <value_statement> <condition_operator> <value_statement>
<condition_operator> ::= "<" | ">" | "<=" | ">=" | "==" | "and" | "or" | "xor"

<assignment_expression> ::= <type> <variable> "=" <value_statement>
<arithmetic_expression> ::= <value_statement> <operator> <value_statement>

<operator> ::= <low_presedence_operator> | <high_presedence_operator>
<low_presedence_operator> ::= "-" | "+"
<high_presedence_operator> ::= "/" | "*"
